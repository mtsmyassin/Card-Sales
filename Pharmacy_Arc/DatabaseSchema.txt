-- FARMACIA CARIMAS DATABASE SCHEMA --

-- 1. USERS TABLE
-- Holds login credentials and access levels.
create table users (
  username text primary key,
  password text not null,
  role text not null,  -- 'staff', 'manager', 'admin', 'super_admin'
  store text           -- 'Carimas #1', 'Carimas #2', 'Carthage', 'All'
);

-- 2. AUDITS TABLE
-- Holds all sales data, payouts, and day-lock records.
create table audits (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  date text not null,       -- YYYY-MM-DD
  store text not null,
  reg text not null,        -- 'Reg 1' or 'DAY_CLOSE'
  staff text,
  gross numeric default 0,
  net numeric default 0,
  variance numeric default 0,
  payload jsonb             -- Stores breakdown, audit trail, approval status
);

-- 3. INDEXES (Optional for speed)
create index idx_audits_date on audits(date);
create index idx_audits_store on audits(store);
